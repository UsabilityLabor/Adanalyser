0011         function subjects=createSubjects(self,config,videoLength)
0012             eegFilePaths = config.EEGFiles;
0013             edaFilePaths = config.EDAFiles;
0014             ecgFilePaths = config.ECGFiles;
0015             numberOfSubjects =  length(edaFilePaths);  
0016             message = ['Reading raw data for ' num2str(numberOfSubjects) ' subject(s)'];
0017             bar = waitbar(0,message);  % open waitbar dialog with message
0018             subjects=cell(1,numberOfSubjects);
0019             for i = 1:numberOfSubjects
0020                 % get subject name and electrode name from eeg file name
0021                 edaFileForSubject = edaFilePaths{i};
0022                 [~,name,~] = fileparts(edaFileForSubject);
0023                 splitFileName =  textscan(name,'%s','Delimiter','_');
0024                 subjectName = splitFileName{1}{2};
0025                 % get eeg files for subject by subject name
0026                 matches = strfind(eegFilePaths,subjectName);
0027                 eegFileIndicies = find(~cellfun(@isempty,matches));
0028                 subject = Subject();
0029                 for j = 1:length(eegFileIndicies)
0030                    eegFileForSubject = eegFilePaths{eegFileIndicies(j)}; 
0031                    subject.eegValuesForElectrodes{j} = self.parseEEGFile(eegFileForSubject,videoLength);
0032                 end
0033                 % get eda file for subject by subject name
0034                 matches = strfind(edaFilePaths,subjectName);
0035                 edaFileIndex = ~cellfun(@isempty,matches);
0036                 edaFileForSubject = edaFilePaths{edaFileIndex};
0037                 % get ecg file for subject by subject name
0038                 matches = strfind(ecgFilePaths,subjectName);
0039                 ecgFileIndex = ~cellfun(@isempty,matches);
0040                 ecgFileForSubject = ecgFilePaths{ecgFileIndex};
0041                 % create the subject
0042                 subject.name = subjectName;
0043                 subject.edaValues = self.parseEDAGFile(edaFileForSubject,videoLength);
0044                 subject.ecgValues = self.parseECGFile(ecgFileForSubject);    
0045                 subjects{i}=subject; 
0046                 % update waitbar
0047                 waitbar(i /numberOfSubjects);
0048             end
0049             close(bar);
